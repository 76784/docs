Handlebars.registerHelper('progress', function (name, percent, stalled) {
  var barWidth = percent / 5
  var bar = "********************".slice(0,barWidth)            
  return bar + " " + percent + "% " + name + " " +  (stalled ? "stalled" : "")
})
