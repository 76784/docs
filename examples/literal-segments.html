Template
{{people.[0].firstname}} {{people.[0].lastname}} lives in {{people.[0].[resident-in]}}
Input
{
  people: [
    {
      firstname: "Yehuda",
      lastname: "Katz",
      "resident-in": "San Francisco"
    },
    {
      firstname: "Nils",
      lastname: "Knappmeier",
      "resident-in": "Darmstadt"
    }
  ]
}
Output
Yehuda Katz lives in San Francisco
