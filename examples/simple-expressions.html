Template
<div class="city">
  <h1>{{name}}</h1>
  <div class="body">
    {{summary}}
  </div>
</div>
Input
{
  name: "Darmstadt",
  summary: "Darmstadt is a city in Hesse, Germany"
}
Output
<div class="city">
  <h1>Darmstadt</h1>
  <div class="body">
    Darmstadt is a city in Hesse, Germany
  </div>
</div>
